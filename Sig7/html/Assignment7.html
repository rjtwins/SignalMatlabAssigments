8.3

%We are goig to reuse vars so clear the worksspace.
clearvars;
%Load data
load('tfr-data.mat');

%Do fourier transform on data
fftData = fft(data, [], 2);
fftAmp = abs(fftData)./size(fftData,2);
n = size(fftData,2);
fs = n/abs(min(time) - max(time));
f = fs/n*(0:n-1);
f_nyquist = fs/2;

%Cutoff at nyquist
if mod(n,2)==0
    n_cutoff = n/2;
else
    n_cutoff=(n-1)/2;
end
f = f(1:n_cutoff);
fftAmp = 2*fftAmp(:,1:n_cutoff);
fftAmp(:,1) = fftAmp(:,1)./2;

ampMean = mean(fftAmp,1);

%Plot both meaned data and multiplied meaned data.
figure(8)
hold on;
title("Fig8 mean of frequency amplitudes")
plot(f,ampMean,'k.-');
xlabel("Frequency,(Hz)");
ylabel("Amplitude");
hold off;

%Calc multiplied means
multMean = ampMean.*f;
figure(9)
hold on;
title("Fig9 multiplied mean of frequency amplitudes")
plot(f,multMean, 'k.-');
xlabel("Frequency(Hz)");
ylabel("Amplitude");
hold off;

8.8

%Init matrix for frequency samples
tfr = zeros(60, size(data,2));
%Loop over frequencies getting samples and filling the matrix.
for i = 1:60
    %The window size must be 3 times the lenght of a period of the
    %frequency.
    period = 1/i;
    samplePeriod = 1/fs;
    wsize = round((period/samplePeriod) * 3);
    tfr(i,:) = get_amplitude(data, fs, wsize, i);
end
%Plot the time, frequency, amplitude image
hold on;
figure(14)
image(32 + 32 * tfr./max(max(abs(tfr))));
title("Fig15 Amplitude of frequencies over time.");
set(gca, 'YDir', 'normal');
set(gca, 'XTick', linspace(1,size(data,2), 10));
tf = time(1);
tl = time(end);
set(gca, 'XtickLabel', round(10 * (linspace(tf,	tl,	10))) / 10);
colormap jet;
xlabel("Time(s)");
ylabel("Frequency");
hold off

8.9

%Get basline data points
baselineData = data(:,time < 0);
%Init a baslineMeans matrix
baselineMeans = zeros(60,900);
%Loop over frequencies and fill rows with means (we fill the entire row
%with one number for easy matrix substractions.
for i = 1:60
    %The window size must be 3 times the lenght of a period of the
    %frequency.
    period = 1/i;
    samplePeriod = 1/fs;
    wsize = round((period/samplePeriod) * 3);
    baselineMeans(i,:) = nanmean(get_amplitude(baselineData, fs, wsize, i));
end
%Substract baseline from frequency samples
cortrf = tfr - baselineMeans;

%Plot corrected time, frequency, amplitude image.
hold on;
figure(15)
image(32 + 32 * cortrf./max(max(abs(cortrf))));
title("Fig15 Corrected amplitude of frequencies over time.");
set(gca, 'YDir', 'normal');
set(gca, 'XTick', linspace(1,size(data,2), 10));
tf = time(1);
tl = time(end);
set(gca, 'XtickLabel', round(10 * (linspace(tf,	tl,	10))) / 10);

xlabel("Time(s)");
ylabel("Frequency");
hold off
